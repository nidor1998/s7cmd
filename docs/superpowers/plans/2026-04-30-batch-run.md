# `batch-run` Subcommand Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `batch-run` subcommand that reads s7cmd commands from stdin and runs them in-process (sequential or parallel, fail-fast or continue-on-error, with progress bar in read-all mode).

**Architecture:** A new `src/batch_run/` module owns parsing, validation, execution, progress, and summary. A new `src/dispatch.rs` extracts the per-subcommand match arms from `main.rs` into a single `dispatch(cmd) -> i32` function reusable by both `main` (single subcommand) and `batch_run` (per-line). Vendored `sync_bin`/`clean_bin`/`ls_bin::run` functions are patched to return `Result<i32>` instead of calling `std::process::exit` from inside.

**Tech Stack:** Rust 2024, clap 4.6 (derive), tokio 1.52 (full), indicatif 0.18, shlex 1, clap-verbosity-flag (already a transitive dep), `std::io::IsTerminal` (no new crate), `std::thread::available_parallelism` (no new crate).

```rust
//! Tokenize lines from stdin into argv-style command vectors,
//! ready to feed into `Cli::try_parse_from`.

use anyhow::{Result, anyhow};
use std::io::BufRead;

/// One parsed line: line number (1-based), the original raw text, and
/// the tokenized argv (with `"s7cmd"` as argv[0]).
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ParsedLine {
    pub line_no: usize,
    pub raw: String,
    pub argv: Vec<String>,
}

/// Tokenize a single line. Returns:
///   - `Ok(Some(argv))` if the line is a real command (with `"s7cmd"` prepended)
///   - `Ok(None)` if the line is blank or a comment
///   - `Err` if shlex rejects it (unbalanced quotes, etc.)
pub fn tokenize_line(line: &str) -> Result<Option<Vec<String>>> {
    let trimmed = line.trim();
    if trimmed.is_empty() || trimmed.starts_with('#') {
        return Ok(None);
    }
    let tokens = shlex::split(trimmed)
        .ok_or_else(|| anyhow!("malformed quoting"))?;
    if tokens.is_empty() {
        return Ok(None);
    }
    let mut argv = Vec::with_capacity(tokens.len() + 1);
    argv.push("s7cmd".to_string());
    argv.extend(tokens);
    Ok(Some(argv))
}

/// Read the whole reader, returning a Vec of ParsedLine. Errors bail out
/// with line number and raw line included in the error message.
pub fn read_all<R: BufRead>(reader: R) -> Result<Vec<ParsedLine>> {
    let mut out = Vec::new();
    for (idx, line_result) in reader.lines().enumerate() {
        let line_no = idx + 1;
        let line = line_result.map_err(|e| anyhow!("read error at line {line_no}: {e}"))?;
        match tokenize_line(&line) {
            Ok(Some(argv)) => out.push(ParsedLine {
                line_no,
                raw: line,
                argv,
            }),
            Ok(None) => continue,
            Err(e) => {
                return Err(anyhow!(
                    "parse error at line {line_no}: {e}\n  > {line}"
                ));
            }
        }
    }
    Ok(out)
}
```

```rust
        })?;
        if matches!(cp_config.source, StoragePath::Stdio)
            || matches!(cp_config.target, StoragePath::Stdio)
        {
            return Err(anyhow!(
                "line {line_no}: stdin/stdout transfers are not allowed inside batch-run\n  > {raw}"
            ));
        }
    }
    if let Cmd::Mv(args) = cmd {
        let mv_config = s3util_rs::Config::try_from(args.clone()).map_err(|e| {
            anyhow!("line {line_no}: {e}\n  > {raw}")
        })?;
        if matches!(mv_config.source, StoragePath::Stdio)
            || matches!(mv_config.target, StoragePath::Stdio)
        {
            return Err(anyhow!(
                "line {line_no}: stdin/stdout transfers are not allowed inside batch-run\n  > {raw}"
            ));
        }
    }
    reject_per_line_tracing(line_no, raw, cmd)?;
    Ok(())
}

/// Tracing flags belong to `batch-run` itself, not to per-line subcommands.
/// Reject any line that sets a tracing flag.
fn reject_per_line_tracing(line_no: usize, raw: &str, cmd: &Cmd) -> Result<()> {
    let (json, aws_sdk, span_events, disable_color) = match cmd {
        Cmd::Cp(a) => (
            a.common.json_tracing,
            a.common.aws_sdk_tracing,
            a.common.span_events_tracing,
            a.common.disable_color_tracing,
        ),
        Cmd::Mv(a) => (
            a.common.json_tracing,
            a.common.aws_sdk_tracing,
            a.common.span_events_tracing,
            a.common.disable_color_tracing,
        ),
        Cmd::Rm(a) => (
            a.common.json_tracing,
            a.common.aws_sdk_tracing,
            a.common.span_events_tracing,
            a.common.disable_color_tracing,
        ),
        // The remaining bucket-* / object-* subcommands all have
        // `args.common: CommonClientArgs` with the same field shape.
        // Group them under one helper to avoid 30+ identical arms.
        other => return reject_via_common(line_no, raw, other),
    };
    if json || aws_sdk || span_events || disable_color {
        return Err(tracing_error(line_no, raw));
    }
    Ok(())
}

fn reject_via_common(line_no: usize, raw: &str, cmd: &Cmd) -> Result<()> {
    macro_rules! check {
        ($a:expr) => {{
            let c = &$a.common;
            if c.json_tracing || c.aws_sdk_tracing
                || c.span_events_tracing || c.disable_color_tracing
            {
                return Err(tracing_error(line_no, raw));
            }
        }};
    }
    match cmd {
        // Already handled above:
        Cmd::Cp(_) | Cmd::Mv(_) | Cmd::Rm(_) | Cmd::BatchRun(_) => {}

        // Sync / Ls / Clean have differently shaped tracing config
        // (their own CLI structs from upstream libs). Validate via
        // their own tracing config builder instead of `common`.
        Cmd::Sync(boxed) => {
            let tracing = boxed.common.json_tracing
                || boxed.common.aws_sdk_tracing
                || boxed.common.span_events_tracing
                || boxed.common.disable_color_tracing;
            if tracing { return Err(tracing_error(line_no, raw)); }
        }
        Cmd::Ls(boxed) => {
            // s3ls-rs CLIArgs has its own tracing fields under .common.
            let tracing = boxed.common.json_tracing
                || boxed.common.aws_sdk_tracing
                || boxed.common.span_events_tracing
                || boxed.common.disable_color_tracing;
            if tracing { return Err(tracing_error(line_no, raw)); }
        }
        Cmd::Clean(boxed) => {
            let tracing = boxed.common.json_tracing
                || boxed.common.aws_sdk_tracing
                || boxed.common.span_events_tracing
                || boxed.common.disable_color_tracing;
            if tracing { return Err(tracing_error(line_no, raw)); }
        }

        // Bucket / object metadata subcommands (CommonClientArgs).
        Cmd::CreateBucket(a) => check!(a),
        Cmd::DeleteBucket(a) => check!(a),
        Cmd::HeadBucket(a) => check!(a),
        Cmd::HeadObject(a) => check!(a),
        Cmd::GetObjectTagging(a) => check!(a),
        Cmd::PutObjectTagging(a) => check!(a),
        Cmd::DeleteObjectTagging(a) => check!(a),
        Cmd::GetBucketTagging(a) => check!(a),
        Cmd::PutBucketTagging(a) => check!(a),
        Cmd::DeleteBucketTagging(a) => check!(a),
        Cmd::GetBucketPolicy(a) => check!(a),
        Cmd::PutBucketPolicy(a) => check!(a),
        Cmd::DeleteBucketPolicy(a) => check!(a),
        Cmd::GetBucketVersioning(a) => check!(a),
        Cmd::PutBucketVersioning(a) => check!(a),
        Cmd::GetBucketLifecycleConfiguration(a) => check!(a),
        Cmd::PutBucketLifecycleConfiguration(a) => check!(a),
        Cmd::DeleteBucketLifecycleConfiguration(a) => check!(a),
        Cmd::GetBucketEncryption(a) => check!(a),
        Cmd::PutBucketEncryption(a) => check!(a),
        Cmd::DeleteBucketEncryption(a) => check!(a),
        Cmd::GetBucketCors(a) => check!(a),
        Cmd::PutBucketCors(a) => check!(a),
        Cmd::DeleteBucketCors(a) => check!(a),
        Cmd::GetPublicAccessBlock(a) => check!(a),
        Cmd::PutPublicAccessBlock(a) => check!(a),
        Cmd::DeletePublicAccessBlock(a) => check!(a),
        Cmd::GetBucketWebsite(a) => check!(a),
        Cmd::PutBucketWebsite(a) => check!(a),
        Cmd::DeleteBucketWebsite(a) => check!(a),
        Cmd::GetBucketLogging(a) => check!(a),
        Cmd::PutBucketLogging(a) => check!(a),
        Cmd::GetBucketNotificationConfiguration(a) => check!(a),
        Cmd::PutBucketNotificationConfiguration(a) => check!(a),
    }
    Ok(())
}
```

```rust
//! Progress bar wrapper around indicatif. Only drawn in read-all mode
//! when stderr is a TTY and `--quiet` is not set.

use indicatif::{ProgressBar, ProgressStyle};
use std::io::IsTerminal;

pub struct Progress {
    bar: Option<ProgressBar>,
    ok: u64,
    failed: u64,
}

impl Progress {
    /// Build a new bar. If progress should not be shown (streaming mode,
    /// `--quiet`, or non-TTY stderr), return a Progress with no bar.
    pub fn new(total: u64, show: bool) -> Self {
        if !show {
            return Self {
                bar: None,
                ok: 0,
                failed: 0,
            };
        }
        let bar = ProgressBar::new(total);
        bar.set_style(
            ProgressStyle::with_template(
                "[{bar:40.cyan/blue}] {pos}/{len} ({msg}) {elapsed_precise}",
            )
            .expect("hard-coded template")
            .progress_chars("=> "),
        );
        bar.set_message("0 ok, 0 failed");
        Self {
            bar: Some(bar),
            ok: 0,
            failed: 0,
        }
    }

    /// Decide whether to draw the bar. Returns true only when
    /// quiet=false, streaming=false, AND stderr is a TTY.
    pub fn should_show(quiet: bool, streaming: bool) -> bool {
        if quiet || streaming {
            return false;
        }
        std::io::stderr().is_terminal()
    }

    /// Record a completion (success or failure) and update the message.
    pub fn tick(&mut self, exit_code: i32) {
        if exit_code == 0 {
            self.ok += 1;
        } else {
            self.failed += 1;
        }
        if let Some(bar) = &self.bar {
            bar.set_message(format!("{} ok, {} failed", self.ok, self.failed));
            bar.inc(1);
        }
    }

    pub fn finish(&self) {
        if let Some(bar) = &self.bar {
            bar.finish();
        }
    }
}
```

```rust
//! Sequential and parallel execution loops.

use crate::batch_run::progress::Progress;
use crate::batch_run::summary::Summary;
use crate::cli::Cmd;

use std::future::Future;
use std::pin::Pin;
use std::sync::Arc;
use std::sync::atomic::{AtomicBool, Ordering};
use std::time::Instant;

use tokio::sync::Semaphore;
use tokio::task::JoinSet;

pub type DispatchFn = Arc<
    dyn Fn(Cmd) -> Pin<Box<dyn Future<Output = i32> + Send>> + Send + Sync,
>;

pub struct PreparedLine {
    pub line_no: usize,
    pub raw: String,
    pub cmd: Cmd,
}

#[derive(Debug, Clone, Copy)]
pub struct ExecutorOptions {
    pub workers: usize,           // resolved (1 = sequential)
    pub continue_on_error: bool,
    pub quiet: bool,
    pub streaming: bool,
}

/// Shared interrupt flag. Set by the SIGINT handler installed in
/// `batch_run::run`; checked by both executors to stop spawning new
/// commands. Per-subcommand cancellation tokens (registered inside
/// each dispatched command) handle propagation into in-flight transfers.
pub type Interrupt = Arc<AtomicBool>;

pub async fn run_sequential(
    lines: Vec<PreparedLine>,
    opts: ExecutorOptions,
    dispatch: DispatchFn,
    interrupt: Interrupt,
) -> (i32, Summary) {
    let total = lines.len() as u64;
    let mut progress = Progress::new(total, Progress::should_show(opts.quiet, opts.streaming));
    let mut summary = Summary::default();
    let start = Instant::now();
    let mut worst = 0i32;

    for (idx, line) in lines.into_iter().enumerate() {
        // Bail out if SIGINT arrived between commands (regardless of
        // continue_on_error — interrupt is unconditional).
        if interrupt.load(Ordering::SeqCst) {
            let processed = idx as u64;
            summary.skipped = total.saturating_sub(processed);
            break;
        }
        eprintln!("==> line {}: {}", line.line_no, line.raw);
        let code = dispatch(line.cmd).await;
        progress.tick(code);
        summary.record(code);
        if code > worst {
            worst = code;
        }
        if code != 0 && !opts.continue_on_error {
            // Skipped count = total - already-processed.
            let processed = (idx + 1) as u64;
            summary.skipped = total.saturating_sub(processed);
            break;
        }
    }

    progress.finish();
    summary.elapsed = start.elapsed();
    (worst, summary)
}

pub async fn run_parallel(
    lines: Vec<PreparedLine>,
    opts: ExecutorOptions,
    dispatch: DispatchFn,
    interrupt: Interrupt,
) -> (i32, Summary) {
    let total = lines.len() as u64;
    let progress = Arc::new(tokio::sync::Mutex::new(Progress::new(
        total,
        Progress::should_show(opts.quiet, opts.streaming),
    )));
    let mut summary = Summary::default();
    let start = Instant::now();

    let sem = Arc::new(Semaphore::new(opts.workers));
    let fail_cancel = Arc::new(AtomicBool::new(false));
    let mut joinset = JoinSet::new();
    let mut spawned = 0u64;

    for line in lines {
        // SIGINT (interrupt) unconditionally stops spawning.
        if interrupt.load(Ordering::SeqCst) {
            break;
        }
        // Failure-driven cancel only stops spawning when fail-fast is on.
        if fail_cancel.load(Ordering::SeqCst) && !opts.continue_on_error {
            break;
        }
        let permit = sem.clone().acquire_owned().await.expect("sem closed");
        let dispatch = Arc::clone(&dispatch);
        let progress = Arc::clone(&progress);
        let fail_cancel = Arc::clone(&fail_cancel);
        let continue_on_error = opts.continue_on_error;
        spawned += 1;
        joinset.spawn(async move {
            let _permit = permit;
            eprintln!("==> line {}: {}", line.line_no, line.raw);
            let code = dispatch(line.cmd).await;
            if code != 0 && !continue_on_error {
                fail_cancel.store(true, Ordering::SeqCst);
            }
            progress.lock().await.tick(code);
            (line.line_no, code)
        });
    }

    let mut worst = 0i32;
    while let Some(joined) = joinset.join_next().await {
        match joined {
            Ok((_, code)) => {
                summary.record(code);
                if code > worst {
                    worst = code;
                }
            }
            Err(e) => {
                tracing::error!(error = format!("{e:#}"), "task panicked");
                summary.record(1);
                worst = worst.max(1);
            }
        }
    }
    summary.skipped = total.saturating_sub(spawned);
    progress.lock().await.finish();
    summary.elapsed = start.elapsed();
    (worst, summary)
}

/// Resolve `--parallel 0` to `num_cpus`. Otherwise pass through.
pub fn resolve_workers(parallel: usize) -> usize {
    if parallel == 0 {
        std::thread::available_parallelism()
            .map(|n| n.get())
            .unwrap_or(1)
    } else {
        parallel
    }
}
```

```rust
//! `batch-run` subcommand.

use anyhow::Result;
use clap::Parser;
use std::sync::Arc;

pub mod args;
pub mod executor;
pub mod parser;
pub mod progress;
pub mod summary;
pub mod validate;

use std::sync::atomic::{AtomicBool, Ordering};

use crate::cli::{BatchRunArgs, Cli};
use executor::{DispatchFn, ExecutorOptions, Interrupt, PreparedLine, resolve_workers};

pub async fn run(args: BatchRunArgs) -> i32 {
    // Install a single SIGINT listener for the whole batch run.
    // It sets the shared flag the executors check before each spawn.
    // (Per-subcommand cancellation tokens — installed inside each
    // dispatched command — handle propagation into in-flight transfers.)
    let interrupt: Interrupt = Arc::new(AtomicBool::new(false));
    {
        let interrupt = Arc::clone(&interrupt);
        tokio::spawn(async move {
            if tokio::signal::ctrl_c().await.is_ok() {
                interrupt.store(true, Ordering::SeqCst);
            }
        });
    }

    // Read stdin and tokenize.
    let stdin = std::io::stdin();
    let parsed = match parser::read_all(stdin.lock()) {
        Ok(p) => p,
        Err(e) => {
            eprintln!("batch-run: {e}");
            return 1;
        }
    };

    // Parse each line into a Cmd via clap, then validate.
    let prepared = match parse_and_validate(parsed) {
        Ok(p) => p,
        Err(e) => {
            eprintln!("batch-run: {e}");
            return 1;
        }
    };

    let workers = resolve_workers(args.parallel);
    let opts = ExecutorOptions {
        workers,
        continue_on_error: args.continue_on_error,
        quiet: args.quiet,
        streaming: args.streaming,
    };

    let dispatch: DispatchFn = Arc::new(|cmd| {
        Box::pin(async move { crate::dispatch::dispatch(cmd).await })
    });

    let (code, summary) = if workers == 1 {
        executor::run_sequential(prepared, opts, dispatch, interrupt).await
    } else {
        executor::run_parallel(prepared, opts, dispatch, interrupt).await
    };

    if !args.quiet {
        eprintln!("{}", summary.format());
    }

    code
}

fn parse_and_validate(parsed: Vec<parser::ParsedLine>) -> Result<Vec<PreparedLine>> {
    let mut prepared = Vec::with_capacity(parsed.len());
    for line in parsed {
        // clap parse: turn argv into Cli, take .command.
        let cli = Cli::try_parse_from(&line.argv).map_err(|e| {
            anyhow::anyhow!(
                "line {}: parse error: {}\n  > {}",
                line.line_no,
                e,
                line.raw
            )
        })?;
        let cmd = cli
            .command
            .ok_or_else(|| {
                anyhow::anyhow!(
                    "line {}: empty command\n  > {}",
                    line.line_no,
                    line.raw
                )
            })?;
        validate::validate(line.line_no, &line.raw, &cmd)?;
        prepared.push(PreparedLine {
            line_no: line.line_no,
            raw: line.raw,
            cmd,
        });
    }
    Ok(prepared)
}
```
